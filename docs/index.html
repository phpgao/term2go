API Preview

Dead simple to use

Connect, traverse the window hierarchy, and send text — all in a few lines.

main.go connection.go 
package main
import "github.com/phpgao/term2go"
func main() {
  // Connect to iTerm2 — auto-detects cookie/key
  term2go.Run(ctx, "my-app", func(caller term2go.Caller) error {
    app, _ := term2go.GetApp(caller)
    // Traverse window → tab → session hierarchy
    for _, w := range app.Windows {
      for _, t := range w.Tabs {
        // Send a command to each session
        sess := t.Root.FindSessionByID("session-id")
        sess.SendText(ctx, "echo Hello term2go!\n", false)
      }
    }
    return nil
  })
}
